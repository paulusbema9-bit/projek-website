                             if (r === "Kasubag TU" || (v.tujuanAkhir && v.tujuanAkhir.includes(r))) act += `<button onclick="bukaTL('${id}')" class="btn btn-sm btn-info w-100 mb-1 text-white">Tindak Lanjut</button>`;
                        }
                        if (v.linkFile) act += `<a href="${v.linkFile}" target="_blank" class="btn btn-sm btn-outline-primary w-100 mb-1">File</a>`;
                        tb.innerHTML += `<tr><td><span class="badge bg-secondary mb-1">${v.category.replace(/_/g,' ')}</span><br><div class="box-nomor">ND: ${lND}</div><br><div class="box-nomor">ST: ${lST}</div><br><strong>${v.perihal}</strong></td><td><span class="badge status-badge ${isS?'bg-success':'bg-warning text-dark'}">${sRaw.toUpperCase()}</span></td><td><div class="p-2 border rounded bg-light">${act}</div></td></tr>`;
                    }
                    if (currentMode === 'rekapND' && (r === "Admin" || r === "Sekretariat")) rb.innerHTML += `<tr><td>${dO.toLocaleDateString()}</td><td>${lND}</td><td>${v.perihal}</td><td>${v.tujuanAkhir.join(', ')}</td><td>${v.pic}</td><td>${sRaw}</td></tr>`;
                    if (currentMode === 'rekapST' && (r === "Admin" || r === "Sekretariat") && v.noUrutST !== "N/A") rb.innerHTML += `<tr><td>${dO.toLocaleDateString()}</td><td>${lST}</td><td>${v.perihal}</td><td>${v.tujuanAkhir.join(', ')}</td><td>${v.pic}</td><td>${sRaw}</td></tr>`;
                });
            });
        }

        window.prosesDisposisi = async () => { 
            const sel = Array.from(document.querySelectorAll('.chk-d:checked')).map(c=>c.value); 
            if(sel.length===0) return alert("Pilih tujuan disposisi!"); 
            const btn = document.getElementById('btnSubmitDispo');
            const file = document.getElementById('fileDispoSekre').files[0];
            btn.disabled = true;
            btn.innerText = "Mengirim...";
            try {
                let updateData = { status: "Selesai Disposisi", tujuanAkhir: sel };
                if (file) {
                    const sRef = ref(storage, 'disposisi/' + Date.now() + "_" + file.name);
                    await uploadBytes(sRef, file);
                    const url = await getDownloadURL(sRef);
                    updateData.linkFile = url;
                }
                await updateDoc(doc(db, "surat_dinas", window.targetDispId), updateData);
                alert("Disposisi Berhasil!"); 
                tutupModal('modalDisposisi');
                document.getElementById('fileDispoSekre').value = "";

                alert("Gagal mengirim disposisi!");

            }
